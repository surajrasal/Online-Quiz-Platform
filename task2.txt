    <button class="option" onclick="checkAnswer(this)"></button>
    <button class="option" onclick="checkAnswer(this)"></button>
    <button class="option" onclick="checkAnswer(this)"></button>
    <button class="option" onclick="checkAnswer(this)"></button>

    <p id="result"></p>
    <button id="nextBtn" onclick="nextQuestion()">Next</button>
</div>

<script>
    const quizData = [
        {
            question: "What is HTML?",
            options: ["Programming", "Markup Language", "Database", "OS"],
            answer: "Markup Language"
        },
        {
            question: "CSS is used for?",
            options: ["Logic", "Design", "Storage", "Server"],
            answer: "Design"
        },
        {
            question: "JavaScript works on?",
            options: ["Frontend", "Backend", "Both", "None"],
            answer: "Both"
        },
        {
            question: "Which is database?",
            options: ["MySQL", "HTML", "CSS", "JS"],
            answer: "MySQL"
        }
    ];

    let current = 0;
    let score = 0;

    function loadQuestion() {
        document.getElementById("result").innerText = "";
        let q = quizData[current];
        document.getElementById("question").innerText = q.question;

        let options = document.getElementsByClassName("option");
        for (let i = 0; i < options.length; i++) {
            options[i].innerText = q.options[i];
        }
    }

    function checkAnswer(btn) {
        let selected = btn.innerText;
        let correct = quizData[current].answer;

        if (selected === correct) {
            document.getElementById("result").innerText = "Correct ✅";
            score++;
        } else {
            document.getElementById("result").innerText = "Wrong ❌";
        }
    }

    function nextQuestion() {
        current++;

        if (current < quizData.length) {
            loadQuestion();
        } else {
            document.querySelector(".container").innerHTML =
                "<h2>Quiz Finished</h2>" +
                "<p>Your Score: " + score + "/" + quizData.length + "</p>";
        }
    }

    loadQuestion();
</script>
